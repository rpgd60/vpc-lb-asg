## curl command to hit ALB

 ```
 export LBURL="app30-alb-774564157.eu-west-1.elb.amazonaws.com"
 while true;  do curl -s -o /dev/null -w "%{url_effective}, %{response_code}, %{time_total}\n" $LBURL ; done
```
or slower
```
 while sleep 1;  do curl -s -o /dev/null -w "%{url_effective}, %{response_code}, %{time_total}\n" $LBURL ; done
```
